105// REALM_LIKELY(expr) - suggest that `expr` is usually true
106// REALM_UNLILELY(expr) - suggest that `expr` is usually false
107// REALM_EXPECT(expr, expval) - suggest that `expr` usually evaluates to `expval`
108#ifdef _MSC_VER
109// no __builtin_expect in MSVC
110#define REALM_LIKELY(expr) (expr)
111#define REALM_UNLIKELY(expr) (expr)
112#define REALM_EXPECT(expr, expval) (expr)
113#else
114#define REALM_LIKELY(expr) __builtin_expect((expr), true)
115#define REALM_UNLIKELY(expr) __builtin_expect((expr), false)
116#define REALM_EXPECT(expr, expval) __builtin_expect((expr), (expval))
117#endif

147// REALM_ATTR_DEPRECATED(msg, thing) - indicates `thing` is deprecated, printing
148// `msg` at compile time if possible
149// REALM_ATTR_DEPRECATED2(msg, thing1, thing2) - above when `thing` has commas in it
150#ifdef __ICC
151#define REALM_ATTR_DEPRECATED(msg, thing) thing __attribute__((deprecated))
152#define REALM_ATTR_DEPRECATED2(msg, thing1, thing2) \
153 thing1, thing2 __attribute__((deprecated))
154#elif defined(_MSC_VER)
155#define REALM_ATTR_DEPRECATED(msg, thing) __declspec(deprecated(msg)) thing
156#define REALM_ATTR_DEPRECATED2(msg, thing1, thing2) \
157 __declspec(deprecated(msg)) thing1, thing2
158#else
159#define REALM_ATTR_DEPRECATED(msg, thing) thing __attribute__((deprecated(msg)))
160#define REALM_ATTR_DEPRECATED2(msg, thing1, thing2) \
161 thing1, thing2 __attribute__((deprecated(msg)))
162#endif

167// REALM_ALIGNED_TYPE_SAMEAS(newtype, origtype, reftype) - defines type `newtype`
168// that is identical to `origtype` except that it is aligned as `reftype`
169#ifdef _MSC_VER
170#define REALM_ALIGNED_TYPE_SAMEAS(newtype, origtype, reftype) \

192// the following defines are used to communicate which declarations are part
193// of the "official" public Realm API (as opposed to things that need to be in
194// the same header files for C++ reasons) - they also control which symbols
195// appear in the shared library:
196// REALM_PUBLIC_API - class/method/variable intended for direct use
197// in applications
198// REALM_INTERNAL_API_EXTERNAL_LINKAGE - method/variable not intended for
199// direct use in applications but used indirectly and
200// therefore requiring external linkage
201// REALM_INTERNAL_API - method/variable that is not part of the public API
202// despite being in an otherwise-public class
203
204#ifdef REALM_LIMIT_SYMBOL_VISIBILITY
205// TODO(cperry): replace this with realm_exports.h generated by cmake
206#if defined(_MSC_VER)
207#ifndef REALM_EXPORT
208#ifdef realm_EXPORTS
209/* We are building this library */
210#define REALM_PUBLIC_API __declspec(dllexport)
211#else
212/* We are using this library */
213#define REALM_PUBLIC_API __declspec(dllimport)
214#endif
215#endif
216#define REALM_INTERNAL_API_EXTERNAL_LINKAGE REALM_PUBLIC_API
217#define REALM_INTERNAL_API
218#else
219#define REALM_PUBLIC_API __attribute__((visibility("default")))
220#define REALM_INTERNAL_API_EXTERNAL_LINKAGE __attribute__((visibility("default")))
221#define REALM_INTERNAL_API __attribute__((visibility("hidden")))
222#endif
223#else
224#define REALM_PUBLIC_API

226#define REALM_INTERNAL_API
