224 template <typename K, typename V, typename LT = Mutex>
225 class LockedMap {
226 public:
227 bool exists(const K &key) const
228 {
229 AutoLock<LT> al(mutex);
230 typename std::map<K, V>::const_iterator it = map.find(key);
231 return (it != map.end());
232 }
233
234 bool put(const K &key, const V &value, bool replace = false)
235 {
236 AutoLock<LT> al(mutex);
237 typename std::map<K, V>::iterator it = map.find(key);
238 if(it != map.end()) {
239 if(replace)
240 it->second = value;
241 return true;
242 } else {
243 map.insert(std::make_pair(key, value));
244 return false;
245 }
246 }
247
248 V get(const K &key, const V &defval) const
249 {
250 AutoLock<LT> al(mutex);
251 typename std::map<K, V>::const_iterator it = map.find(key);
252 if(it != map.end())
253 return it->second;
254 else
255 return defval;
256 }
257
258 // protected:
259 mutable LT mutex;
260 std::map<K, V> map;
261 };
