61 enum
62 {
63 SRC_PARTIAL_OK = (1 << 0),
64 SRC_LINES_OK = (1 << 1),
65 SRC_PLANES_OK = (1 << 2),
66 SRC_FLAGMASK = 0xff,

68 DST_PARTIAL_OK = (1 << 8),
69 DST_LINES_OK = (1 << 9),
70 DST_PLANES_OK = (1 << 10),
71 DST_FLAGMASK = 0xff00,
72
73 PARTIAL_OK = SRC_PARTIAL_OK | DST_PARTIAL_OK,
74 LINES_OK = SRC_LINES_OK | DST_LINES_OK,
75 PLANES_OK = SRC_PLANES_OK | DST_PLANES_OK,
