98 struct DeferredSpawnCache {
99 static const size_t MAX_ENTRIES = 4;
100 Mutex mutex;
101 EventImpl *events[MAX_ENTRIES];
102 EventImpl::gen_t generations[MAX_ENTRIES];
103 Task *tasks[MAX_ENTRIES];
104 size_t counts[MAX_ENTRIES];
105 size_t ages[MAX_ENTRIES];
106 size_t current_age;
107
108 void clear()
109 {
110 current_age = 0;
111 memset(events, 0, sizeof(events));
112 memset(tasks, 0, sizeof(tasks));
113 memset(counts, 0, sizeof(counts));
114 memset(ages, 0, sizeof(ages));
115 memset(generations, 0, sizeof(generations));
116 }
117
118 void flush()
119 {
120 for(size_t i = 0; i < MAX_ENTRIES; i++) {
121 if(tasks[i] != nullptr) {
122 tasks[i]->remove_reference();
123 }
124 }
125 clear();
126 }
127 };
