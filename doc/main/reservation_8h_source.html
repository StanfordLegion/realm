64 // a FastReservation is a wrapper around a Reservation that signficantly
65 // reduces overhead for repeated use of the same Reservation within a
66 // (coherent) address space. Unlike Reservation's, a FastReservation is
67 // an opaque object that is not copyable or transferrable. However, a
68 // FastReservation _does_ play nice with other users of the underlying
69 // Reservation, on the same node or on others
70
71 class REALM_PUBLIC_API FastReservation {
72 public:
73 FastReservation(Reservation _rsrv = Reservation::NO_RESERVATION);
74 ~FastReservation(void);
75
76 private:
77 // NOT copyable
78 FastReservation(const FastReservation&);
79 FastReservation& operator=(const FastReservation&);
80
81 public:
82 enum WaitMode {
83 SPIN, // keep trying until lock is available
84 ALWAYS_SPIN, // spin, even if holder is suspended (DANGEROUS)
85 WAIT, // wait on Realm::Event, allowing other tasks to run
86 EXTERNAL_WAIT, // wait on kernel mutex, for non-Realm threads
87 };
88
89 // these are inlined and cover the fast (i.e. uncontended) cases - they
90 // automatically fall back to the ..._slow versions below when needed
91 // if the lock cannot be obtained without waiting, these calls return an
92 // Event that the caller should wait on before trying again
93 // Note #1: use exists() (not has_triggered()!) to check if the acquisition
94 // failed and needs to be retried
95 // Note #2: unlike try_acquire above, the caller is NOT required to try again
96 Event lock(WaitMode mode = SPIN); // synonym for wrlock()
97 Event wrlock(WaitMode mode = SPIN);
98 Event rdlock(WaitMode mode = SPIN);
99 void unlock(void);
100
101 // non-blocking versions return true if the lock is granted, false if
102 // waiting/spinning would have been required
103 // Note: when a FastReservation has a backing Reservation, these calls can
104 // fail even if no other thread holds the FastReservation - in such cases,
105 // the only recourse is to keep attempting the acquisition until it
106 // succeeds (this can be problematic if the holder needs the same processor)
107 bool trylock(void); // synonym for trywrlock()
108 bool trywrlock(void);
109 bool tryrdlock(void);
110
111 // in general, a thread holding a lock should not go to sleep on other
112 // events, as another thread waiting on the lock may spin and prevent the
113 // holding thread from running on that core
114 // to avoid this, a thread holding the lock that intends to sleep should
115 // create a UserEvent and advise on entry/exit from the region in which
116 // sleeping might occur (do not provide the event the holder is waiting
117 // for - the user event must be triggered after the advise_sleep_exit call)
118 void advise_sleep_entry(UserEvent guard_event);
119 void advise_sleep_exit(void);
120
121 // the fast path stores several things in a 32-bit word that can be
122 // atomically updated
123 typedef uint32_t State;
124 static const State STATE_READER_COUNT_MASK = 0x03ffffff;
125 static const State STATE_SLEEPER = 0x04000000;
126 static const State STATE_WRITER = 0x08000000;
127 static const State STATE_WRITER_WAITING = 0x10000000;
128 static const State STATE_BASE_RSRV = 0x20000000;
129 static const State STATE_BASE_RSRV_WAITING = 0x40000000;
130 static const State STATE_SLOW_FALLBACK = 0x80000000;
131
132 protected:
133 Event wrlock_slow(WaitMode mode);
134 Event rdlock_slow(WaitMode mode);
135 void unlock_slow(void);
136 bool trywrlock_slow(void);
137 bool tryrdlock_slow(void);
138
139 friend struct FastRsrvState;
140
141 // we will make use of atomics for the fast path, so make sure we take
142 // a full cache line for our data to avoid false sharing
143 static const size_t CACHE_LINE_SIZE = 64;
144 REALM_ALIGNED_TYPE_CONST(State_aligned, atomic<State>, 16);
145 State_aligned state;
146 // this is slightly fragile, but we want to have enough room to store
147 // the implementation-specific stuff without another layer of
148 // indirection
149 char opaque[CACHE_LINE_SIZE * 4 - sizeof(atomic<State>)];
150 };
